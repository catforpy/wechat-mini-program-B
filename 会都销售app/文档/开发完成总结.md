### 2. 首页类目选择系统
**文件位置**：[pages/index/index.vue](pages/index/index.vue)

**功能结构**：
```
首页
├── 搜索框（简洁设计，无图标）
├── 主体类型选择（企业/个人/境外）
├── 快速入口（4个Logo卡片）
├── 一级类目（横向滚动标签）
├── 二级类目（左侧垂直列表）
└── 三级模板（右侧卡片网格）
```

**关键代码**：
```typescript
// 主体类型切换
const entityTypes: EntityTypeOption[] = [
  { name: '企业', type: EntityType.COMPANY },
  { name: '个人', type: EntityType.PERSONAL },
  { name: '境外', type: EntityType.OVERSEAS }
]

// 根据主体类型获取类目
const currentCategories = computed(() => {
  const entityType = entityTypes[currentEntityType.value].type
  return getCategoriesByEntityType(entityType)
})
```

#### 数据结构
```typescript
// 三级类目模板
interface TemplateInfo {
  name: string      // 模板名称
  desc: string      // 简短描述
}

// 二级类目
interface SecondLevelCategory {
  name: string                    // 二级类目名称
  qualification?: string          // 资质要求
  scope?: string                  // 适用范围
  templates: TemplateInfo[]       // 三级类目（小程序模板）
}

// 一级类目
interface FirstLevelCategory {
  name: string                      // 一级类目名称
  hasData: boolean                 // 是否有具体数据
  secondLevel: SecondLevelCategory[] // 二级类目列表
}

// 主体类型
enum EntityType {
  COMPANY = 'company',      // 企业/政府等非个人主体
  PERSONAL = 'personal',    // 个人主体
  OVERSEAS = 'overseas'     // 境外主体
}
```
